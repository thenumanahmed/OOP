#include<iostream>

using namespace std;

class Student{
	int age;
public: 
	Student(){
		cout << "Hi from constructor" << endl;
	}
	~Student(){
		cout << "Fi AmaniLLah!";
	}

	Student(Student &obj){
		age = obj.age;
	}

	void setAge(int a){
		age = a;
	}
	int getAge(){
		return age;
	}
};

int main(){
	{
		Student std1;
		std1.setAge(20);
		cout << "the age of std1 is : " << std1.getAge() << endl;
		Student std2(std1);
		cout << "the age of std2 is : " << std2.getAge() << endl;
	}
	cout << "I am out!";
	
	return 0;
}

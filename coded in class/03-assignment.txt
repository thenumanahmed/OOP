#include<iostream>
using namespace std;
class Student{
	int *rollNumber;
public:
	Student(int r=0){
		rollNumber = new int();
		*rollNumber = r;
	}

	Student(const Student &obj){
		rollNumber = new int();
		*rollNumber = *(obj.rollNumber);
	}
	~Student(){
		cout << " deleting rollNumber"<<endl;
		delete rollNumber;
	}
	void operator = (const Student &obj){
		*rollNumber = *(obj.rollNumber);
	}
	int getRollNumber(){
		return *rollNumber;
	}
	void setRollNumber(int i){
		*rollNumber = i;
	}
	
};

int main(){
	{
		Student std1(101);
		Student std2(std1);

		Student std3;

		std3 = std2;

		cout << std1.getRollNumber() << endl;
		cout << std2.getRollNumber() << endl;

		std2.setRollNumber(106);

		cout << std1.getRollNumber() << endl;
		cout << std2.getRollNumber() << endl;
		cout << std3.getRollNumber() << endl;

	}
	
	return 0;
}

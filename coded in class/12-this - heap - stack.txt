#include<iostream>
using namespace std;

class MyClass{
public:
	int x; 
	int y;
};


int main(){
	MyClass myObject; //defined on the stack memory
	MyClass *ptr = new MyClass();//defined on heap
	// dynamic memory allocation
	myObject.x = 10;
	cout << myObject.x << endl;
	ptr->x = 20;
	cout << ptr->x << endl;
}

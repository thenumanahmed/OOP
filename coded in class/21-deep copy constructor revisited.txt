// C++ program for required implementation
#include <iostream>
using namespace std;
class MyClass{
	int *x;
public:
	MyClass(int i=0){
		x = new int();
		*x = i;
	}
	MyClass(MyClass& o){
		cout << "hi from copy constructor" << endl;
		x = new int();
		*x = *(o.x);
	}
	int getX(){
		return *x;
	}
	void setX(int i){
		*x = i;
	}
	~MyClass(){
		cout << "Bye from the destructor"<<endl;
	}
};

int main()
{
	MyClass obj1(5);
	MyClass obj2(obj1);
	cout << obj1.getX() << endl; //5
	cout << obj2.getX() << endl; //5 
	obj1.setX(10);
	cout << obj1.getX() << endl; //10
	cout << obj2.getX() << endl; //10



	return 0;
}
